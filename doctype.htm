    document.getElementById('year').textContent = new Date().getFullYear();

 

    // Mobile nav toggle

    const toggle = document.getElementById('toggleNav');

    const mainNav = document.getElementById('mainNav');

    toggle && toggle.addEventListener('click', () => {

      const expanded = toggle.getAttribute('aria-expanded') === 'true';

      toggle.setAttribute('aria-expanded', String(!expanded));

      if (!expanded) {

        mainNav.style.display = 'flex';

        mainNav.style.flexDirection = 'column';

        mainNav.style.gap = '8px';

        mainNav.style.padding = '12px';

      } else {

        mainNav.style.display = '';

      }

    });

 

    // Smooth scrolling for anchors

    document.querySelectorAll('a[href^="#"]').forEach(a => {

      a.addEventListener('click', function(e){

        const target = document.querySelector(this.getAttribute('href'));

        if(target){ e.preventDefault(); target.scrollIntoView({behavior:'smooth', block:'start'}); }

      });

    });

 

    // Contact form: Netlify-ready + mailto fallback

    const form = document.getElementById('contactForm');

    const feedback = document.getElementById('formFeedback');

    const resetBtn = document.getElementById('resetBtn');

 

    resetBtn && resetBtn.addEventListener('click', () => form.reset());

 

    form.addEventListener('submit', function(e){

      // If hosted on Netlify with forms enabled, Netlify will handle submission.

      // Here we also provide a mailto fallback for simple deployments (opens user's email client).

      // We detect Netlify by checking data-netlify attribute; still provide fallback to avoid lost messages.

      e.preventDefault();

      feedback.textContent = 'Enviando...';

 

      const name = form.querySelector('[name=name]').value.trim();

      const email = form.querySelector('[name=email]').value.trim();

      const message = form.querySelector('[name=message]').value.trim();

 

      if(!name || !email || !message){

        feedback.textContent = 'Por favor, preencha todos os campos.';

        return;

      }

 

      // Try to submit via fetch (works if backend or Netlify form endpoint exists)

      const formData = new FormData(form);

 

      // Attempt fetch post to current URL - will succeed on Netlify if configured.

      fetch(window.location.pathname, {

        method: 'POST',

        body: formData,

        headers: {'Accept': 'application/json'}

      }).then(res => {

        if(res.ok){

          feedback.textContent = 'Mensagem enviada com sucesso. Obrigado!';

          form.reset();

        } else {

          // fallback: open mailto

          const subject = encodeURIComponent('Contato via site - ' + name);

          const body = encodeURIComponent('Nome: ' + name + '\\nEmail: ' + email + '\\n\\n' + message);

          window.location.href = 'mailto:seu-email@exemplo.com?subject=' + subject + '&body=' + body;

          feedback.textContent = 'Abrindo seu cliente de email para envio (fallback).';

        }

      }).catch(err => {

        // fallback mailto

        const subject = encodeURIComponent('Contato via site - ' + name);

        const body = encodeURIComponent('Nome: ' + name + '\\nEmail: ' + email + '\\n\\n' + message);

        window.location.href = 'mailto:seu-email@exemplo.com?subject=' + subject + '&body=' + body;

        feedback.textContent = 'Abrindo seu cliente de email para envio (fallback).';

      });

    });

 

    // Accessibility: focus outline visible on keyboard navigation

    (function(){

      function handleFirstTab(e) {

        if (e.key === 'Tab') {

          document.body.classList.add('user-is-tabbing');

          window.removeEventListener('keydown', handleFirstTab);

        }

      }

      window.addEventListener('keydown', handleFirstTab);

    })();
